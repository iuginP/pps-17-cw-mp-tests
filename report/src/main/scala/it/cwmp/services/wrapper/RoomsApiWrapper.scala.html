14 /**
15   * A trait that describes the Api wrapper for Rooms
16   *
17   * @author Enrico Siboni
18   */
19 trait RoomsApiWrapper {
20 
21   /**
22     * Creates a room
23     *
24     * @param roomName      the room name
25     * @param playersNumber the players number
26     * @param userToken     the token to be authenticated against api
27     * @return the future containing the identifier of the created room,
28     *         or fails if roomName is empty or playersNumber not correct
29     */
30   def createRoom(roomName: String, playersNumber: Int)(implicit userToken: String): Future[String]
31 
32   /**
33     * Enters a room
34     *
35     * @param roomID              the identifier of the room
36     * @param userAddress         the address of the paler that wants to enter
37     * @param notificationAddress the address where player wants to receive notification of room filling
38     * @param userToken           the token to be authenticated against api
39     * @return the future that completes when the user has entered,
40     *         or fails if roomID not provided, not present
41     *         or user already inside a room, or room full
42     */
43   def enterRoom(roomID: String, userAddress: Address, notificationAddress: Address)(implicit userToken: String): Future[Unit]
44 
45   /**
46     * Retrieves room information
47     *
48     * @param roomID    the identifier of the room
49     * @param userToken the token to be authenticated against api
50     * @return the future that completes when the room information is available,
51     *         or fails if room id not provided or not present
52     */
53   def roomInfo(roomID: String)(implicit userToken: String): Future[Room]
54 
55   /**
56     * Exits a room
57     *
58     * @param roomID    the identifier of the room to exit
59     * @param userToken the token to be authenticated against api
60     * @return the future that completes when user has exited,
61     *         or fails if roomId is not provided or not present
62     *         or user is not inside that room
63     */
64   def exitRoom(roomID: String)(implicit userToken: String): Future[Unit]
65 
66   /**
67     * retrieves a list of available public rooms
68     *
69     * @param userToken the token to be authenticated against api
70     * @return a future that completes when such list is available
71     */
72   def listPublicRooms()(implicit userToken: String): Future[Seq[Room]]
73 
74   /**
75     * Enters a public room
76     *
77     * @param playersNumber       the number of players that the public room has to have
78     * @param userAddress         the address of the paler that wants to enter
79     * @param notificationAddress the address where player wants to receive notification of room filling
80     * @param userToken           the token to be authenticated against api
81     * @return the future that completes when user ha entered,
82     *         or fails if players number is not correct,
83     *         or user already inside a room
84     */
85   def enterPublicRoom(playersNumber: Int, userAddress: Address, notificationAddress: Address)(implicit userToken: String): Future[Unit]
86 
87   /**
88     * Retrieves information about a public room with specific number of players
89     *
90     * @param playersNumber the number of players that the public room has to have
91     * @param userToken     the token to be authenticated against api
92     * @return the future that completes when the information is available,
93     *         or fails if players number not correct
94     */
95   def publicRoomInfo(playersNumber: Int)(implicit userToken: String): Future[Room]
96 
97   /**
98     * Exits a public room
99     *
100     * @param playersNumber the number of players that the public room has to have
101     * @param userToken     the token to be authenticated against api
102     * @return the future taht completes when the user has exited,
103     *         or fails if players number is not correct,
104     *         or user not inside that room
105     */
106   def exitPublicRoom(playersNumber: Int)(implicit userToken: String): Future[Unit]
107 }

128   /**
129     * Implementation of the Api Wrapper for rooms service
130     *
131     * @param clientOptions the implicit configuration for the client connection to the host
132     *
133     */
134   private class RoomsApiWrapperDefault(override protected val clientOptions: WebClientOptions)
135     extends RoomsApiWrapper with RoomApiWrapperUtils with VertxInstance with VertxClient {
136 
137     override def createRoom(roomName: String, playersNumber: Int)(implicit userToken: String): Future[String] =
138       createPrivateRoomRequest(roomName, playersNumber)
139         .flatMap(implicit response => handleResponse(Future.successful(response.bodyAsString().get), 201))
140 
141     override def enterRoom(roomID: String, userAddress: Address, notificationAddress: Address)(implicit userToken: String): Future[Unit] =
142       enterPrivateRoomRequest(roomID, userAddress, notificationAddress)
143         .flatMap(implicit response => handleResponse(Future.successful(Unit), 200))
144 
145     override def roomInfo(roomID: String)(implicit userToken: String): Future[Room] =
146       privateRoomInfoRequest(roomID)
147         .flatMap(implicit response => handleResponse({
148           import Room.Converters._
149           Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom)
150         }, 200))
151 
152     override def exitRoom(roomID: String)(implicit userToken: String): Future[Unit] =
153       exitPrivateRoomRequest(roomID)
154         .flatMap(implicit response => handleResponse(Future.successful(Unit), 200))
155 
156     override def listPublicRooms()(implicit userToken: String): Future[Seq[Room]] =
157       listPublicRoomsRequest()
158         .flatMap(implicit response => handleResponse({
159           Future.successful {
160             import Room.Converters._
161             Json.fromArrayString(response.bodyAsString().get)
162               .stream().toArray().toSeq.map(_.toString)
163               .map(jsonString => Json.fromObjectString(jsonString).toRoom)
164           }
165         }, 200))
166 
167     /**
168       * Utility method to handle the Service response
169       */
170     private def handleResponse[T](onSuccessFuture: => Future[T], successHttpCodes: Int*)(implicit response: HttpResponse[Buffer]) =
171       successHttpCodes find (_ == response.statusCode) match {
172         case Some(_) => onSuccessFuture
173         case None => response.bodyAsString match {
174           case Some(body) =>
175             Future.failed(HTTPException(response.statusCode, body))
176           case None =>
177             Future.failed(HTTPException(response.statusCode))
178         }
179       }
180 
181     override def enterPublicRoom(playersNumber: Int, userAddress: Address, notificationAddress: Address)(implicit userToken: String): Future[Unit] =
182       enterPublicRoomRequest(playersNumber, userAddress, notificationAddress)
183         .flatMap(implicit response => handleResponse(Future.successful(Unit), 200))
184 
185     override def publicRoomInfo(playersNumber: Int)(implicit userToken: String): Future[Room] =
186       publicRoomInfoRequest(playersNumber)
187         .flatMap(implicit response => handleResponse({
188           import Room.Converters._
189           Future.successful(Json.fromObjectString(response.bodyAsString().get).toRoom)
190         }, 200))
191 
192     override def exitPublicRoom(playersNumber: Int)(implicit userToken: String): Future[Unit] =
193       exitPublicRoomRequest(playersNumber)
194         .flatMap(implicit response => handleResponse(Future.successful(Unit), 200))
195   }
