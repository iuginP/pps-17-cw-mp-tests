52     * Handles the creation of a private room
53     */
54   private def createPrivateRoomHandler: Handler[RoutingContext] = implicit routingContext => {
55     log.info("Create Private Room Request received...")
56     request.bodyHandler(body =>
57       request.checkAuthenticationOrReject.map(_ =>
58         extractIncomingRoomFromBody(body) match {
59           case Some((roomName, neededPlayers)) =>
60             daoFuture.map(_.createRoom(roomName, neededPlayers)
61               .onComplete {
62                 case Success(generatedID) => sendResponse(201, generatedID)
63                 case Failure(ex) => sendResponse(400, ex.getMessage)
64               })
65           case None =>
66             log.warn("Request body for room creation is required!")
67             sendResponse(400, s"$INVALID_PARAMETER_ERROR no Room JSON in body")
68         }))
69 
70     def extractIncomingRoomFromBody(body: Buffer): Option[(String, Int)] = {
71       try {
72         val jsonObject = body.toJsonObject
73         if ((jsonObject containsKey Room.FIELD_NAME) && (jsonObject containsKey Room.FIELD_NEEDED_PLAYERS))
74           Some((jsonObject getString Room.FIELD_NAME, jsonObject getInteger Room.FIELD_NEEDED_PLAYERS))
75         else None
76       } catch {
77         case _: Throwable => None
78       }
79     }
80   }

82   /**
83     * Handles entering in a private room
84     */
85   private def enterPrivateRoomHandler: Handler[RoutingContext] = implicit routingContext => {
86     log.info("Enter Private Room Request received...")
87     request.bodyHandler(body =>
88       request.checkAuthenticationOrReject.map(user => {
89         (getRequestParameter(Room.FIELD_IDENTIFIER), extractAddressesFromBody(body)) match {
90           case (Some(roomID), Some(addresses)) =>
91             daoFuture.map(implicit dao => dao.enterRoom(roomID)(Participant(user.username, addresses._1.address), addresses._2).onComplete {
92               case Success(_) => handlePrivateRoomFilling(roomID)
93                 .transform({
94                   case Success(_) => Failure(new Exception())
95                   case Failure(_: NoSuchElementException) => Success(Unit)
96                   case Failure(ex) => log.error("Error handling Private Room Filling", ex); Success(Unit)
97                 })
98                 .map(_ => sendResponse(200))
99               case Failure(ex: NoSuchElementException) => sendResponse(404, ex.getMessage)
100               case Failure(ex) => sendResponse(400, ex.getMessage)
101             })
102 
103           case _ =>
104             log.warn("Player and notification address are required for entering a room!")
105             sendResponse(400, s"$INVALID_PARAMETER_ERROR ${Room.FIELD_IDENTIFIER} or ${Address.FIELD_ADDRESS}")
106         }
107       }))
108   }
