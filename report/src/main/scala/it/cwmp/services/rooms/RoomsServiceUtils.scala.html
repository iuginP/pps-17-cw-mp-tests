99                                              (implicit communicationStrategy: RoomReceiverApiWrapper,
100                                               executionContext: ExecutionContext): Future[Unit] = {
101     log.info(s"Preparing to send participant list to room ${roomInformation._1.name} (with id:${roomInformation._1.identifier}) participants ...")
102     val notificationAddresses = for (notificationAddress <- roomInformation._2) yield notificationAddress
103     val players = for (player <- roomInformation._1.participants) yield player
104     log.info(s"Participant notification addresses to contact: $notificationAddresses")
105     log.info(s"Information to send -> $players")
106     Future.sequence {
107       notificationAddresses map (notificationAddress => communicationStrategy.sendParticipants(notificationAddress.address, players))
108     } map (_ => Unit)
