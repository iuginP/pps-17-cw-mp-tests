13     * @param point1 the first point

15     * @return the distance

17   def distance(point1: Point, point2: Point): Double = {
18     Math.sqrt {
19       (point1.x.toDouble - point2.x).squared +
20         (point1.y.toDouble - point2.y).squared
21     }
22   }

24   /**
25     * Calculates the angular coefficient of a straight line passing through two points
26     *
27     * m in this formula -> y = mx + q
28     *
29     * @param point1 the first point
30     * @param point2 the second point
31     * @return the angular coefficient or [[Double.PositiveInfinity]] if the straight line is vertical
32     */
33   def angularCoefficient(point1: Point, point2: Point): Double = {
34     if (point1.x == point2.x) Double.PositiveInfinity
35     else (point1.y.toDouble - point2.y) / (point1.x.toDouble - point2.x)
36   }

38   /**
39     * Calculates the ordinate at the origin of a straight line passing through two points
40     *
41     * q in this formula -> y = mx + q
42     *
43     * @param point1 the first point
44     * @param point2 the second point
45     * @return the ordinate at the origin or [[Double.NaN]] if the straight line is vertical
46     */
47   def ordinateAtOrigin(point1: Point, point2: Point): Double = {
48     if (point1.x == point2.x) Double.NaN
49     else (point1.x.toDouble * point2.y - point2.x.toDouble * point1.y) / (point1.x.toDouble - point2.x)
50   }

52   /**
53     * Calculates the delta X and Y starting from point1 of a point that will be on a straight line passing through two point at given distance from first point
54     *
55     * @param point1   the point from which to refer for the distance
56     * @param point2   the other point from which the straight line should pass
57     * @param distance the distance from point1 on the straight line towards point2
58     * @return a couple of numbers that represents the delta X and Y from point1
59     *         where will be placed a point at given distance on the straight line
60     */
61   def deltaXYFromFirstPoint(point1: Point,
62                             point2: Point,
63                             distance: Double): (Double, Double) = {
64     require(distance > 0, "Distance should be greater that 0")
65 
66     val deltaY =
67       if (point1.y == point2.y) 0 // point on same horizontal line has no delta Y
68       else if (point1.x == point2.x) distance * Math.signum(point2.y.toLong - point1.y) // point on same vertical line has delta Y that equals distance to travel; signum function makes delta positive if first point below second, negative otherwise
69       else distance / Math.sqrt(1 / angularCoefficient(point1, point2).squared + 1) * Math.signum(point2.y.toLong - point1.y) // this is the formula to get deltaY from a distance, the signum function has same significance as in above comment
70 
71     val deltaX =
72       if (point1.x == point2.x) 0 // point on same vertical line has no delta X
73       else if (point1.y == point2.y) distance * Math.signum(point2.x.toLong - point1.x) // point on same horizontal line has delta X that equals distance to travel; signum function makes delta positive if first point is on the left of the second, negative otherwise
74       else distance / Math.sqrt(angularCoefficient(point1, point2).squared + 1) * Math.signum(point2.x.toLong - point1.x) // this is the formula to get deltaX from a distance, the signum function has same significance as in above comment
75 
76     (deltaX, deltaY)
77   }
