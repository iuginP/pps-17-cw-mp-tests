46       * Removes the tentacle from the world refunding for energy the attacker
47       *
48       * @param tentacle the tentacle to remove
49       * @return the modified world
50       */
51     def --(tentacle: Tentacle): CellWorld = {
52       val tentacleActualLength = tentacle.length(cellWorld.instant)
53       val energyRefund = lengthToEnergyReductionStrategy(tentacleActualLength)
54       val attackerAndOthersPair = cellWorld.characters.partition(Cell.ownerAndPositionMatch(_, tentacle.from))
55       val attackerCell = attackerAndOthersPair._1.head
56       val attackerRefunded = Cell(attackerCell.owner, attackerCell.position, attackerCell.energy + energyRefund)
57       CellWorld(cellWorld.instant, attackerAndOthersPair._2 :+ attackerRefunded, cellWorld.attacks filterNot (_ == tentacle))
58     }
59   }

61   /**

66   /**

69   val ATTACK_DURATION_TO_ENERGY_REDUCTION_RATE = 1000d

72   /**

77   val lengthToEnergyReductionStrategy: SizingStrategy[Long, Double] =

79 
80   /**

84     */
85   val durationToEnergyConversionStrategy: SizingStrategy[Duration, Double] =
