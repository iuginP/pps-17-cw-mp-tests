212   private def enterRoom(): Future[Address] = {
213     logger.debug(s"Starting the local one-time reception server...")
214     // Apre il server in ricezione per la lista dei partecipanti
215     listenForParticipantListFuture(
216       // Quando ha ricevuto la lista dei partecipanti dal server
217       participants => {
218         logger.info(s"Participants list received!")
219         becomeInGame(participants)
220       }
221     ).andThen({ // Una volta creato
222       case Success(address) =>
223         logger.debug(s"Server completely started listening at the address: $address")
224       case Failure(error) => // Invia un messaggio di errore alla GUI
225         logger.error(s"Problem starting the server: ${error.getMessage}")
226         roomViewActor ! AlertMessages.Error("Error", error.getMessage)
227     })
