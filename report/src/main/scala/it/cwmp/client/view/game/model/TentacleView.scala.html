32   def reachedPoint(tentacle: Tentacle, actualInstant: Instant): Point = {

34       val tentacleActualLength = tentacle.length(actualInstant)
35       val attackerPosition = tentacle.from.position
36       val deltaXYFromAttackerPosition = GeometricUtils.deltaXYFromFirstPoint(attackerPosition, tentacle.to.position, tentacleActualLength)

38         attackerPosition.y + deltaXYFromAttackerPosition._2.toInt)
39     } else tentacle.to.position
